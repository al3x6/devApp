--------------------Inserer un nouvel utilisateur-------------
INSERT INTO `utilisateur` (`login`, `email`, `mdp`) VALUES ($login, $email, $mdp);
--pour inserer dans la base de donner à partir des données entrer dans le formulaire

--------------------Changer mot de passe-------------
UPDATE `utilisateur` SET `mdp` = '$mdp' WHERE `utilisateur`.`login` = '$login';
--On change l'ancien mot de passe par le nouveau mot de passe d'un utilisateur
--dans le formulaire changement de mot de passe qui aura un mot de passe changer avec
--la confirmation du mot de passe changé

--------------------Insertion d'une visite du site-------------

--------------------Insertion d'une connexion d'un utilisateur dans un module-------------

--------------------Voir les utilisateurs inscrits-------------
SELECT login FROM `utilisateur`;

--------------------Supprimer un utilisateur de la base de données-------------
DELETE FROM `utilisateur` WHERE login = '$login';
